char data="0";
char js[10];

void setup(){
  Serial.begin(9600);
  pinMode(13,OUTPUT);
  }
void loop(){
     int p=0;
    while(Serial.available()>0){
      data=Serial.read();
      js[p]=data;
      p++;
      delay(2);
      }
      if(js[0]=='1')
      digitalWrite(13,HIGH);
      if(js[0]=='1')
      digitalWrite(13,LOW);
      delay(1000);
    }
